\begin{frame}
      \frametitle{Model Counting}
      \begin{block}{Definition: Unweighted Model Counting}
            Given a Boolean formula $\pf$:
            \begin{itemize}
                  \item Exact: find $\#\pf$
                  \item Approximate: $\Pr[(1+\epsilon)^{-1}\#\pf \leq A \leq (1+\epsilon)\#\pf] \geq 1-\delta$
            \end{itemize}
      \end{block}
      \pause
      \begin{block}{Definition: Weighted Model Counting}
            Given $\pf$ and $\wt:\vf{\pf}\mapsto[0,1]$:
            \begin{itemize}
                  \item Weight of $\pf$: sum of the weights of the satisfying assignments
                        \begin{itemize}
                              \item Weight of $x$: $\wt(x)$
                              \item Weight of $\lnot x$: $1-\wt(x)$
                              \item Weight of $\as$: product of the weights of the individual literals
                        \end{itemize}
            \end{itemize}
      \end{block}
\end{frame}
